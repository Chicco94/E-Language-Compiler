def factorial(n :int){
    if (n == 1) {
        return 1
    } else if (n > 0) {
        return n * factorial(n-1)
    } else {
        // secondo test
        // throw("invalid argument to factorial"+n) 
        throw("invalid argument to factorial")
  }
}
