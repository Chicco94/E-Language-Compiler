    def factorial(n :(int>0)) :int{

def factorial(n :int) :int{
    if (n == 1) {
        return 1;
    } else if (n > 0) {
        // espressioni come argomenti 
        return n * factorial(n-1);
    } else {
        throw("invalid argument to factorial"+n) ;
  }
}

def test_function(array_of_tests, function) :bool{
    for test in array_of_tests{
        print( function(test) );
    }
    return True;
}

def main() :int{
    def tests :int := 20;
    test_function(0..test, factorial);
    return 1;
}
