/** manca gaurdia di tipo
    def factorial(n :(int>0)) :int{
*/ 
def factorial(n :int) :int{
    if (n == 1) {
        return 1;
    } else if (n > 0) {
        // espressioni come argomenti 
        return n * factorial(n-1);
    } else {
        throw("invalid argument to factorial"+n) ;
  }
}

def main() :int{
    def tests :int := 0..20;
    for test in tests{
        print( factorial(test) );
    }
    return 1;
}
