def factorial(n :int ) :int{
    if (n == 1) {
        return 1;
    } else if (n > 0) {
        return n * factorial(n);
        // return n * factorial(n-1);
    } else {
        def message :string := "invalid argument to factorial";
        throw(message);
        //throw("invalid argument to factorial"); 
        //throw("invalid argument to factorial"+n);        
  }
}
